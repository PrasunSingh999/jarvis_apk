// netlify/functions/jarvis-chat.js // // Server-side proxy for the xAI Grok API (OpenAI-compatible chat completions). // Keeps your API key out of client-side code entirely. // // SETUP: // 1. In the Netlify dashboard: Site settings → Environment variables // Add XAI_API_KEY = xai-... (your real Grok key from console.x.ai) // 2. Deploy. Netlify auto-detects anything in netlify/functions/. // 3. The function will be live at: https://.netlify.app/.netlify/functions/jarvis-chat const JARVIS_SYSTEM_PROMPT = `You are J.A.R.V.I.S., a hyper-intelligent, unfailingly composed personal assistant in the mold of Tony Stark's AI. Traits: - Extremely capable, proactive, and precise. You anticipate needs and offer next steps before being asked. - Address the user respectfully (sir/ma'am/by name if given) with dry, subtle wit — never goofy, never verbose for its own sake. - Structure substantive answers in four short sections when the question warrants it: 1) High-Level Overview, 2) Deep Dive Analysis, 3) Counterpoints / Challenges, 4) Actionable Next Steps. For simple/quick questions, just answer directly and concisely — don't force the structure. - You understand and fluently respond in ANY language the user writes in. Always reply in the same language the user used, unless asked to translate or switch. - Be talkative and warm but efficient — no filler, no hedging paragraphs. Confident, professional, occasionally clever. - You cannot actually control real-world devices from this browser demo — if asked to do something physical (lights, cars, home automation), acknowledge the request in-character and explain what integration would be needed, rather than pretending it happened.`; exports.handler = async (event) => { const headers = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'Content-Type', 'Access-Control-Allow-Methods': 'POST, OPTIONS', }; if (event.httpMethod === 'OPTIONS') { return { statusCode: 204, headers, body: '' }; } if (event.httpMethod !== 'POST') { return { statusCode: 405, headers, body: JSON.stringify({ error: 'Method not allowed' }) }; } const apiKey = process.env.XAI_API_KEY; if (!apiKey) { return { statusCode: 500, headers, body: JSON.stringify({ error: 'Server misconfigured: XAI_API_KEY not set in Netlify environment variables.' }), }; } let payload; try { payload = JSON.parse(event.body || '{}'); } catch (e) { return { statusCode: 400, headers, body: JSON.stringify({ error: 'Invalid JSON body' }) }; } const messages = Array.isArray(payload.messages) ? payload.messages : []; if (messages.length === 0) { return { statusCode: 400, headers, body: JSON.stringify({ error: 'messages[] is required' }) }; } // Grok's chat completions endpoint is OpenAI-compatible: system prompt goes // inside the messages array as role "system", not as a separate top-level field. const grokMessages = [{ role: 'system', content: JARVIS_SYSTEM_PROMPT }, ...messages]; try { const res = await fetch('https://api.x.ai/v1/chat/completions', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + apiKey, }, body: JSON.stringify({ model: 'grok-4.5', messages: grokMessages, max_tokens: 1000, }), }); const data = await res.json(); if (!res.ok) { return { statusCode: res.status, headers, body: JSON.stringify({ error: data.error ? (data.error.message || data.error) : 'Upstream API error' }), }; } const reply = (data.choices && data.choices[0] && data.choices[0].message && data.choices[0].message.content) || ''; return { statusCode: 200, headers, body: JSON.stringify({ reply: reply.trim() }), }; } catch (err) { return { statusCode: 502, headers, body: JSON.stringify({ error: 'Failed to reach xAI API: ' + err.message }), }; } };
